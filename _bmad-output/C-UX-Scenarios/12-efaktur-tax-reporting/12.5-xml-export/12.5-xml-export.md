## XML Format (DJP Standard)

```xml
<?xml version="1.0" encoding="UTF-8"?>
<FK>
  <FAPR>
    <KD_JENIS_TRANSAKSI>01</KD_JENIS_TRANSAKSI>
    <FG_PENGGANTI>0</FG_PENGGANTI>
    <NOMOR_FAKTUR>00012345</NOMOR_FAKTUR>
    <MASA_PAJAK>05</MASA_PAJAK>
    <TAHUN_PAJAK>2026</TAHUN_PAJAK>
    <TANGGAL_FAKTUR>10/05/2026</TANGGAL_FAKTUR>
    <NPWP>01234567890</NPWP>
    <NAMA>PT Nama Perusahaan</NAMA>
    <ALAMAT_LENGKAP>Jalan...</ALAMAT_LENGKAP>
    <JUMLAH_DPP>15500000</JUMLAH_DPP>
    <JUMLAH_PPN>1860000</JUMLAH_PPN>
    <JUMLAH_PPNBM>0</JUMLAH_PPNBM>
    ...
  </FAPR>
</FK>
```
